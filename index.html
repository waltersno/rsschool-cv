function firstNonRepeatingLetter(s) {
  let finish = s.split("").filter((item, i, arr) => {
    return (
      !arr
        .join("")
        .toLowerCase()
        .split("")
        .slice(i + 1)
        .includes(item.toLowerCase()) && arr.indexOf(item) === i
    );
  })[0];
  return finish ? finish : "";
}
